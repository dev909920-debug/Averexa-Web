// Google Apps Script — Nayi Sheet ke liye
// 1. Ye script form ka data leti hai (JSON POST)
// 2. Sheet me row append karti hai
// 3. Resume ko Google Drive me save karti hai
// 4. dev@ + info@ dono ko email bhejti hai

const TO_EMAILS = ["dev909920@gmail.com", "info@averexaplacement.com"];
const SHEET_NAME = "Sheet1";

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);

    // Handle resume — save to Drive with clean naming
    let resumeLink = "";
    if (data.resumeBase64 && data.resumeMime) {
      const clean = function (str) {
        return str.replace(/\s+/g, "_").replace(/[^a-zA-Z0-9_-]/g, "");
      };
      const safeName = clean(data.fullName) + "_" + clean(data.targetJobTitle) + "_" + Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd");
      const ext = data.resumeName && data.resumeName.includes(".") ? "." + data.resumeName.split(".").pop() : ".pdf";
      const fileName = safeName + ext;

      const blob = Utilities.newBlob(
        Utilities.base64Decode(data.resumeBase64),
        data.resumeMime,
        fileName
      );
      const folder = DriveApp.getFolderById("19uD8wbeKz7ROi7SXv7rtp4I3D7A8wN3Y");
      const file = folder.createFile(blob);
      resumeLink = file.getUrl();
    }

    // Append row to sheet
    sheet.appendRow([
      new Date(),
      data.fullName,
      data.email,
      data.phone,
      data.targetJobTitle,
      data.description || "",
      resumeLink || "Not provided"
    ]);

    // Email notification to BOTH accounts
    const subject = "New Enquiry from " + data.fullName;
    const body =
      "New Contact Form Submission\n\n" +
      "Name: " + data.fullName + "\n" +
      "Email: " + data.email + "\n" +
      "Phone: " + data.phone + "\n" +
      "Target Job: " + data.targetJobTitle + "\n" +
      "Description: " + (data.description || "N/A") + "\n" +
      "Resume: " + (resumeLink || "Not provided") + "\n\n" +
      "Submitted: " + new Date().toLocaleString();

    TO_EMAILS.forEach(function (email) {
      MailApp.sendEmail(email, subject, body);
    });

    return ContentService
      .createTextOutput(JSON.stringify({ success: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: String(err) }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
