// Google Apps Script — Referral Form ke liye
// 1. Ye script referral form ka data leti hai (JSON POST)
// 2. Sheet me row append karti hai
// 3. dev@ + info@ dono ko email bhejti hai
// 4. Phone number +91 ke saath Sheet mein save hota hai

const TO_EMAILS = ["dev909920@gmail.com", "info@averexaplacement.com"];
const SHEET_NAME = "Sheet1";
const PHONE_COLUMN = 7;  // G column — phone number ka column

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);

    // Append row normally
    sheet.appendRow([
      new Date(),
      data.type || "referral",
      data.referrerName,
      data.referrerEmail,
      data.candidateName,
      data.candidateEmail,
      data.candidatePhone || "Not provided",
      data.candidateRole || "Not specified",
      data.notes || "N/A"
    ]);

    // Force phone cell as plain text so + sign stays
    var lastRow = sheet.getLastRow();
    var phoneCell = sheet.getRange(lastRow, PHONE_COLUMN);
    phoneCell.setNumberFormat('@');
    phoneCell.setValue(data.candidatePhone || "Not provided");

    // Email notification to BOTH accounts
    var subject = "New Referral from " + data.referrerName;
    var body = "New Referral Submission\n\n" +
      "Referrer Name: " + data.referrerName + "\n" +
      "Referrer Email: " + data.referrerEmail + "\n" +
      "Candidate Name: " + data.candidateName + "\n" +
      "Candidate Email: " + data.candidateEmail + "\n" +
      "Candidate Phone: " + (data.candidatePhone || "Not provided") + "\n" +
      "Target Role: " + (data.candidateRole || "Not specified") + "\n" +
      "Notes: " + (data.notes || "N/A") + "\n\n" +
      "Submitted: " + new Date().toLocaleString();

    for (var i = 0; i < TO_EMAILS.length; i++) {
      MailApp.sendEmail(TO_EMAILS[i], subject, body);
    }

    return ContentService
      .createTextOutput(JSON.stringify({ success: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: String(err) }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
